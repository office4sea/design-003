lotDataBind()

Description:

  • GraftElement
    ()=> GraftElement.lotDataBindMap
    data-bind 지시자로 등록된 객체를 반환 합니다.

Example
// <div class="form">
//   <input data-bind="input" type="text">
//   <button data-bind="submit" class="btn">확인</button>
//   <div data-bind="output" class="txtBox"></div>
// </div>
const form= graft.tag.div(_it=> {
    // 바인딩된 엘리먼트를 반환 받음
    const vo= _it.lot.dataBind();

    // 버튼 클릭 이벤트 등록
    vo.submit.lot.event('click', e=> {
        // input의 입력값을 output에 출력
        vo.output.lot.text= vo.input.lot.value;
    });
}, {class:'form'}, [
    graft.tag.input({dataBind:'input', type:'text'}),
    graft.tag.button('확인', {dataBind:'submit', class:'btn'}),
    graft.tag.div({dataBind:'output', class:'txtBox'}),
]);

Type:
  • string | Array.<string>
Example

// <input type="text" value="yyy">

// 체크박스 처리
const check1= graft.tag.input({type:'checkbox', name:'check', value:'xxx', checked:'checked'});
const check2= graft.tag.input({type:'checkbox', name:'check', value:'yyy', checked:'checked'});
const form= graft.tag.div([
    check1, check2,
]);

// 출력값: ['xxx', 'yyy']
graft.log.out('value: ', check1.lot.value);
// value="yyy"의 체크박스만 선택됨
check1.lot.value= ['yyy'];
