GraftBridge
네이티브와 웹 매체간의 주고 받는 메시지 구조체 입니다.

Type:
  • object
Example
// Web에서 네이트브로 전달된 메시지 구조
// 네이티브에서 수행할 기능을 command를 이용하여 호출 하며,
// 전달할 인자가 있는 경우 payload를 사용 합니다.
{
    trid: 'w112233',
    command: 'cmd_sample',
    payload: {}
}

// 네이트브에서 Web으로 응답할 결과 메시지 구조
// 웹으로 부터 받은 command 수행 후,
// Graft.postMessage 메소드를 이용하여 수행 완료 처리 합니다.
// 수행완료 시 웹으로 부터 받은 trid와 receive를 조합하여 인자로 전달 합니다.
Graft.postMessage({
    trid: 'w112233',
    receive: {}
});

Type:
  • object
Example
// 네이티브 기능 모깅 처리
graft.bridge.mock.set('cmd_sample', msg=> {
    // 아래 postMessage로 전달한 메시지
    graft.log.out('bridge.mock::cmd_sample', msg);

    // 요청에 대한 결과 반환값
    msg.receive= {
        message: '네이티브 처리 결과 입니다.'
    };

    // 네이티브 수신 메소드를 호출 하여
    // 브릿지를 호출 한 쪽에 트리거 될 수 있도록 한다.
    Graft.postMessage(msg);
});

graft.exec(_=> {
    // 네이티브 기능 요청 메시지 전달
    graft.bridge.postMessage({
        command: 'cmd_sample',
        payload: {xxx: 1}
    })
    .then(receive=> {
        graft.log.out('요청 결과', receive);
    });
});

graft.bridge.setNative(_=> {
    const nativeName= '네이티브 객체 이름';
    // IOS
    if(window.webkit) {
        return window.webkit.messageHandlers[nativeName];
    }
    // AOS
    if(window[nativeName]) {
        return window[nativeName];
    }
});
